# -------------------------------
# Evaluation Function for online_newton_update_meta Strategy (without delta as tuneable)
# -------------------------------
def evaluate_online_newton_update_meta(b, price_relative_vectors, learning_rate, base_experts):
    """
    Runs the online_newton_update_meta strategy with the given hyperparameters and returns performance metrics.
    
    Parameters:
        b: Initial portfolio (numpy array).
        price_relative_vectors: numpy array of price relative vectors (shape: [T, N]).
        learning_rate: Hyperparameter for the learning rate.
        base_experts: List of functions representing the base experts.
        
    Returns:
        A dictionary containing performance metrics.
    """
    # Run the online_newton_update_meta strategy with the provided hyperparameters
    b_n = online_newton_update_meta(b, price_relative_vectors, learning_rate=learning_rate, base_experts=base_experts)
    
    # Calculate performance metrics
    final_wealth = calculate_cumulative_wealth(b_n, price_relative_vectors)
    n_periods = len(price_relative_vectors)
    exp_growth = calculate_exponential_growth_rate(final_wealth, n_periods)
    cum_wealth = calculate_cumulative_wealth_over_time(b_n, price_relative_vectors)
    daily_returns = compute_periodic_returns(cum_wealth)
    sharpe = compute_sharpe_ratio(daily_returns)
    
    return {
        'algorithm': 'online_newton_update_meta',
        'learning_rate': learning_rate,
        'base_experts': str([expert.__name__ for expert in base_experts]),
        'final_wealth': final_wealth,
        'exp_growth': exp_growth,
        'sharpe': sharpe
    }

# -------------------------------
# Wrapper Function for Joblib Evaluation (for online_newton_update_meta)
# -------------------------------
def evaluate_online_newton_combo(param_dict, b, price_relative_vectors_array):
    """
    Wrapper that extracts hyperparameters from the combination dictionary and calls evaluate_online_newton_update_meta.
    """
    try:
        learning_rate = param_dict['learning_rate']
        base_experts = param_dict['base_experts']
        result = evaluate_online_newton_update_meta(b, price_relative_vectors_array, learning_rate, base_experts)
        # Ensure grid parameters are recorded
        result['learning_rate'] = learning_rate
        result['base_experts'] = str([expert.__name__ for expert in base_experts])
        return result
    except Exception as e:
        print(f"‚ùå ERROR: {param_dict} | Exception: {str(e)}")
        return {
            'learning_rate': param_dict.get('learning_rate'),
            'base_experts': str([expert.__name__ for expert in param_dict.get('base_experts', [])]),
            'final_wealth': None,
            'exp_growth': None,
            'sharpe': None,
            'error': str(e)
        }

# -------------------------------
# Generate Valid Parameter Combinations (for online_newton_update_meta)
# -------------------------------
def generate_valid_combinations(grid):
    """
    Generates a list of parameter dictionaries from the grid for online_newton_update_meta.
    """
    valid_combinations = []
    for lr in grid['learning_rate']:
        for experts in grid['base_experts']:
            valid_combinations.append({
                'learning_rate': lr,
                'base_experts': experts
            })
    return valid_combinations

# -------------------------------
# Build Comprehensive Base Experts List
# -------------------------------
# Consider the base algorithms: cwmr, follow_the_regularized_leader, pamr, anticor, 
# along with additional strategies for Online Newton Update: olmar and exponential_gradient.
# In our default online_newton_update_meta, we use [cwmr, olmar, exponential_gradient],
# but for tuning, we allow any combination from a selected pool.
base_algorithms =  [cwmr, follow_the_regularized_leader, pamr]
all_base_expert_combinations = []
for r in range(2, len(base_algorithms) + 1):
    for combo in itertools.combinations(base_algorithms, r):
        all_base_expert_combinations.append(list(combo))

# -------------------------------
# Joblib Grid Search for online_newton_update_meta
# -------------------------------
def run_joblib_grid_online_newton(grid, b, price_relative_vectors_array, n_jobs=4):
    param_dicts = generate_valid_combinations(grid)
    print(f"üîÅ Starting joblib tuning for online_newton_update_meta with {len(param_dicts)} combinations on {n_jobs} workers")
    
    results = Parallel(n_jobs=n_jobs)(
        delayed(evaluate_online_newton_combo)(param_dict, b, price_relative_vectors_array)
        for param_dict in tqdm(param_dicts, desc="Tuning Progress")
    )
    
    results_df = pd.DataFrame(results)
    results_df.sort_values(by='sharpe', ascending=False, inplace=True)
    best_result = results_df.iloc[0].to_dict()
    return results_df, best_result

# -------------------------------
# Define the grid for online_newton_update_meta tuning
# -------------------------------
online_newton_grid = {
    'learning_rate': [.001,0.01],
    'base_experts': all_base_expert_combinations
}

# -------------------------------
# Run the Parallelized Tuning for online_newton_update_meta Strategy
# -------------------------------
if __name__ == '__main__':
    n_jobs = max(1, os.cpu_count() - 2)
    print(f"Detected {os.cpu_count()} logical cores. Using {n_jobs} workers for parallel tuning.")
    
    start = time.time()
    newton_results_df, best_newton = run_joblib_grid_online_newton(online_newton_grid, b, price_relative_vectors_array, n_jobs=n_jobs)
    
    # Save results for future reference
    newton_results_df.to_csv("..\\Data\\Tuning Data\\online_newton_tuning_results.csv", index=False)
